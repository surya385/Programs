package looping_samples;
	import java.util.Scanner;
	public class Spy_Number { // sum and product of the digits are equal - spy number
		public static void main(String args[])  
		{   
	 
		Scanner spy = new Scanner(
System.in
);  
		System.out.print("Enter a number : " );  
		int num=spy.nextInt(); 
		spy.spynumber();
		}
		public void spynumber() {
			int p =1;
			int s=0 ;
			int rem; 
			int num;
		while(num>0)  
			{  
		rem=num%10;  //remainder i.e last digit of the number    // Iteration 1 :   
		s=s+rem;  
		p=p*rem;  
		num=num/10;  // last digit will be removed
		if(s==p)  
			System.out.println("spy numnber");  
			else  
			System.out.println("not a spy numnber");  
			}  
		}
	}
